```javascript
function extractJsonObject(text) {
  const start = text.indexOf("{");
  const end = text.lastIndexOf("}");
  if (start === -1 || end === -1 || end <= start) return null;
  try {
    return JSON.parse(text.slice(start, end + 1));
  } catch {
    return null;
  }
}
```

The script:
1. Accepts a URL as the first CLI argument: `node test-veena-crawl.js https://example.com`
2. Falls back to `https://example.com` if no argument provided
3. Calls `crawlCompanyForMKG(url)` from veena-crawler.js
4. Asserts all 12 TOP_LEVEL_FIELDS are present in the result
5. Logs each field with its confidence score and whether value is null or populated
6. Prints a summary: "PASS: all 12 fields present" or "FAIL: missing fields: [x, y]"
7. Also calls `buildContextPatchFromCrawl(result, "veena", "test-run-001")` and asserts the patch has all 12 fields in MKGService envelope format (has value, confidence, last_verified, source_agent, expires_at keys)
8. Exits with code 0 on pass, code 1 on failure

```javascript
// test-veena-crawl.js
// Usage: node test-veena-crawl.js [url]
// Convention: matches test-supabase-connection.js style (no test framework, plain Node.js)

import { crawlCompanyForMKG, buildContextPatchFromCrawl } from "./platform/content-engine/veena-crawler.js";

const TOP_LEVEL_FIELDS = [
  "positioning", "icp", "competitors", "offers", "messaging",
  "channels", "funnel", "metrics", "baselines", "content_pillars",
  "campaigns", "insights",
];

const url = process.argv[2] || "https://example.com";
console.log(`\n[test-veena-crawl] Testing crawl for: ${url}\n`);

let exitCode = 0;

try {
  console.log("Step 1: crawlCompanyForMKG...");
  const result = await crawlCompanyForMKG(url);

  // Assert all 12 fields present
  const missing = TOP_LEVEL_FIELDS.filter(f => !(f in result));
  if (missing.length > 0) {
    console.error(`FAIL: missing fields: ${missing.join(", ")}`);
    exitCode = 1;
  } else {
    console.log("PASS: all 12 fields present\n");
  }

  // Log each field
  for (const field of TOP_LEVEL_FIELDS) {
    const entry = result[field] || {};
    const filled = entry.value !== null && entry.value !== undefined;
    console.log(`  ${field}: confidence=${entry.confidence ?? "??"} value=${filled ? "populated" : "null"}`);
  }

  console.log("\nStep 2: buildContextPatchFromCrawl...");
  const patch = buildContextPatchFromCrawl(result, "veena", "test-run-001");
  const patchMissing = TOP_LEVEL_FIELDS.filter(f => !(f in patch));
  const envelopeFields = ["value", "confidence", "last_verified", "source_agent", "expires_at"];
  const badEnvelopes = TOP_LEVEL_FIELDS.filter(f => !envelopeFields.every(ef => ef in (patch[f] || {})));

  if (patchMissing.length > 0) {
    console.error(`FAIL: patch missing fields: ${patchMissing.join(", ")}`);
    exitCode = 1;
  } else if (badEnvelopes.length > 0) {
    console.error(`FAIL: patch envelope missing keys for: ${badEnvelopes.join(", ")}`);
    exitCode = 1;
  } else {
    console.log("PASS: context_patch covers all 12 fields with correct envelope format");
  }

} catch (err) {
  console.error("FAIL: unexpected error:", err.message);
  exitCode = 1;
}

console.log(`\n[test-veena-crawl] Result: ${exitCode === 0 ? "PASS" : "FAIL"}`);
process.exit(exitCode);
```
